48// Tests that death tests report thrown exceptions as failures and that the
49// exceptions do not escape death test macros.
50TEST(CxxExceptionDeathTest, ExceptionIsFailure) {
51 try {
52 EXPECT_NONFATAL_FAILURE(EXPECT_DEATH(throw 1, ""), "threw an exception");
53 } catch (...) { // NOLINT
54 FAIL() << "An exception escaped a death test macro invocation "
55 << "with catch_exceptions "
56 << (GTEST_FLAG_GET(catch_exceptions) ? "enabled" : "disabled");
57 }
58}
